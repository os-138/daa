function a = mergeSort(a) 
    if length(a) > 1 then 
        m = floor(length(a) / 2); 
        l = a(1:m); 
        r = a(m+1:$); 
        l = mergeSort(l); 
        r = mergeSort(r);

            if l(1) < r(1) then

                l(1) = [];

                a($+1) = r(1);

    end

2. Write a Scilab program to solve the Nuts & Bolts Problem using Quick Sort. 
Answer: 
 
function [n,b] = matchNB(n,b,l,h) 
    if l < h then 
        p = partNB(n,b,l,h); 
        [n,b] = matchNB(n,b,l,p-1); 
        [n,b] = matchNB(n,b,p+1,h); 
    end 
endfunction 
 
function p = partNB(n,b,l,h) 
    pv = n(l); 
    i = l; 
    for j = l+1:h 
        if b(j) < pv then 
            i = i + 1; 
            t = b(i); b(i) = b(j); b(j) = t; 
        elseif b(j) == pv then 
            t = b(j); b(j) = b(l); b(l) = t; 
            j = j - 1; 
        end 
    end 
    t = b(i); b(i) = b(l); b(l) = t; 
    pv = b(i); 
    i2 = l; 
    for j2 = l+1:h 
        if n(j2) < pv then 
            i2 = i2 + 1; 
            t = n(i2); n(i2) = n(j2); n(j2) = t; 
        elseif n(j2) == pv then 
            t = n(j2); n(j2) = n(l); n(l) = t; 
            j2 = j2 - 1; 
        end 
    end 
    t = n(i2); n(i2) = n(l); n(l) = t; 
    p = i2; 
endfunction

function [min, assign] = assignj(c)

    p = perms(1:n);

function f = maze(m, x, y, s) 
    n = size(m, 1); 
    if x == n & y == n then 
        s(x, y) = 1; 
        f = %t; 
        return 
    end 
    if x > 0 & x <= n & y > 0 & y <= n & m(x, y) == 0 & s(x, y) == 0 then 
        s(x, y) = 1; 
        if maze(m, x+1, y, s) then f = %t; return; end 
        if maze(m, x, y+1, s) then f = %t; return; end 
        if maze(m, x-1, y, s) then f = %t; return; end 
        if maze(m, x, y-1, s) then f = %t; return; end 
        s(x, y) = 0; 
    end 
    f = %f; 
endfunction 
 
m = [0 1 0 0; 
     0 0 0 1; 
     1 0 1 0; 
     0 0 0 0]; 
 
s = zeros(4,4); 
if maze(m, 1, 1, s) then 
    disp(s) 
else 
    disp("No path") 
end

1. Given a set of cities and distance between every pair of cities, the problem is 
to find the shortest possible tour that visits every city exactly once and returns 
A: 

function [bestCost, bestPath] = tsp(d, s, v, c, p, bestCost, bestPath)
    n = size(d, 1);
    
    // If all cities are visited
    if size(v, "*") == n then
        c = c + d(s, 1);    // return to start city (1)
        if c < bestCost then
            bestCost = c;
            bestPath = [p 1];
        end
        return;
    end

    // Try all next cities
    for i = 1:n
        if ~or(v == i) then
            [bestCost, bestPath] = tsp( ...
                d, ...                 // distance matrix
                i, ...                 // new current city
                [v i], ...             // updated visited list
                c + d(s, i), ...       // updated cost
                [p i], ...             // updated path
                bestCost, ...          // current best cost
                bestPath ...           // current best path
            );
        end
    end
endfunction

// Distance matrix
d = [0 10 15 20;
     10 0 35 25;
     15 35 0 30;
     20 25 30 0];

// Initialization
bestCost = %inf;
bestPath = [];

// Start from city 1
[bestCost, bestPath] = tsp(d, 1, [1], 0, [1], bestCost, bestPath);

disp("Min cost = " + string(bestCost));
disp("Path = " + string(bestPath));
